Descripción
Principiante WiFi · Diagnóstico

El ESP-32 escanea todas las redes WiFi cercanas y las lista en el Monitor Serie con SSID, intensidad de señal en dBm, canal, tipo de cifrado y una barra visual de señal. El escaneo se repite cada 10 segundos automáticamente. No requiere conectarse a ninguna red.

Componentes
ESP-32 Board1 PC
Cable USB A → C1 PC
Conexiones

Sin conexiones externas — solo el ESP-32 conectado por USB. La salida se ve en el Monitor Serie.

Ejemplo de salida en Monitor Serie
Monitor Serie — 115200 baud

Código Arduino
C++ / Arduino IDE (ESP-32) 
#include <WiFi.h>

void linea(int len = 50) {
  for (int i = 0; i < len; i++) Serial.print('-');
  Serial.println();
}

String cifrado(wifi_auth_mode_t enc) {
  switch (enc) {
    case WIFI_AUTH_OPEN:         return "Abierta";
    case WIFI_AUTH_WEP:          return "WEP";
    case WIFI_AUTH_WPA_PSK:      return "WPA";
    case WIFI_AUTH_WPA2_PSK:     return "WPA2";
    case WIFI_AUTH_WPA_WPA2_PSK: return "WPA/WPA2";
    case WIFI_AUTH_WPA3_PSK:     return "WPA3";
    default:                      return "Otro";
  }
}

String barra(int rssi) {
  if (rssi >= -50) return "████████ Excelente";
  if (rssi >= -60) return "██████░░ Bueno";
  if (rssi >= -70) return "████░░░░ Regular";
  if (rssi >= -80) return "██░░░░░░ Debil";
  return              "░░░░░░░░ Muy debil";
}

void escanear() {
  Serial.println("\n=== Buscando redes WiFi... ===\n");

  int n = WiFi.scanNetworks();

  if (n == 0) {
    Serial.println("No se encontraron redes.");
    return;
  }

  Serial.printf("%-30s %4s  %3s  %-12s  %s\n",
    "SSID", "dBm", "Ch", "Cifrado", "Senal");
  linea();

  for (int i = 0; i < n; i++) {
    String ssid = WiFi.SSID(i);
    if (ssid.length() > 28) ssid = ssid.substring(0, 26) + "..";

    Serial.printf("%-30s %4d  %3d  %-12s  %s\n",
      ssid.c_str(),
      WiFi.RSSI(i),
      WiFi.channel(i),
      cifrado(WiFi.encryptionType(i)).c_str(),
      barra(WiFi.RSSI(i)).c_str()
    );
  }

  linea();
  Serial.printf("Total: %d redes encontradas\n", n);
  WiFi.scanDelete();
}

void setup() {
  Serial.begin(115200);
  WiFi.mode(WIFI_STA);
  WiFi.disconnect();
  delay(100);
  Serial.println("Escaner WiFi listo.");
}

void loop() {
  escanear();
  Serial.println("\nEsperando 10 segundos...");
  delay(10000);
}
Cómo usar
1
No necesitas modificar nada — el código funciona tal cual.
2
Sube el código al ESP-32 y abre el Monitor Serie a 115200 baud.
3
Verás la lista de redes cercanas. El escaneo se repite cada 10 segundos.
Tip: Cambia el delay(10000) del final del loop() para hacer el escaneo más rápido o lento. Un valor de 5000 escanea cada 5 segundos.
