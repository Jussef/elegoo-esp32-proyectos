Descripción
Avanzado Motores · WiFi/NTP · OLED

Un dispensador de porciones (comida de mascotas, dulces o pastillas) que reparte solo a las horas que programes. El motor paso a paso gira un tambor o tornillo una fracción de vuelta para soltar una ración exacta. El ESP-32 obtiene la hora real por WiFi (NTP), así que las dosis caen puntuales sin necesidad de un reloj externo. La OLED muestra la hora, la próxima dosis programada y cuántas raciones lleva servidas, y un botón permite dispensar una porción manual cuando quieras.

Código Arduino
C++ / Arduino IDE (ESP-32) 
#include <WiFi.h>
#include <time.h>
#include <Wire.h>
#include <Adafruit_GFX.h>
#include <Adafruit_SSD1306.h>
#include <Stepper.h>

// ----- WiFi -----
const char* SSID  = "TU_RED_WIFI";
const char* PASS  = "TU_PASSWORD";

// ----- NTP (zona horaria) -----
const long  GMT_OFFSET   = -5 * 3600;  // ajusta a tu huso (ej. -5 = Colombia/Peru)
const int   DST_OFFSET   = 0;          // horario de verano (0 si no aplica)

// ----- OLED -----
#define ANCHO 128
#define ALTO  64
Adafruit_SSD1306 oled(ANCHO, ALTO, &Wire, -1);

// ----- Motor -----
const int STEPS_REV = 2048;
#define IN1 19
#define IN2 18
#define IN3 5
#define IN4 17
Stepper motor(STEPS_REV, IN1, IN3, IN2, IN4);
const long PASOS_DOSIS = STEPS_REV / 4;   // 1/4 de vuelta por racion (ajustable)

// ----- Boton -----
#define PIN_BTN 4

// ----- Horarios de dosis (hora:minuto) -----
const int HORAS[]   = {8, 14, 20};   // 8:00, 14:00 y 20:00
const int MINUTOS[] = {0, 0,  0};
const int N_DOSIS   = 3;

int  racionesServidas = 0;
int  ultimaDosisMin   = -1;   // evita repetir la dosis dentro del mismo minuto

void apagarBobinas() {
  digitalWrite(IN1, LOW); digitalWrite(IN2, LOW);
  digitalWrite(IN3, LOW); digitalWrite(IN4, LOW);
}

void dispensar(const char* origen) {
  Serial.printf("Dispensando (%s)\n", origen);
  motor.step(PASOS_DOSIS);
  apagarBobinas();
  racionesServidas++;
}

void pantalla(const char* hora, const char* prox) {
  oled.clearDisplay();
  oled.setTextColor(SSD1306_WHITE);
  oled.setTextSize(2);  oled.setCursor(0, 0);   oled.println(hora);
  oled.setTextSize(1);  oled.setCursor(0, 26);  oled.print("Proxima: "); oled.println(prox);
  oled.setCursor(0, 42); oled.print("Raciones: "); oled.println(racionesServidas);
  oled.display();
}

void setup() {
  Serial.begin(115200);
  pinMode(PIN_BTN, INPUT_PULLUP);
  Wire.begin(21, 22);
  oled.begin(SSD1306_SWITCHCAPVCC, 0x3C);
  motor.setSpeed(12);

  oled.clearDisplay();
  oled.setTextSize(1); oled.setCursor(0, 0);
  oled.println("Conectando WiFi..."); oled.display();

  WiFi.begin(SSID, PASS);
  while (WiFi.status() != WL_CONNECTED) { delay(400); Serial.print("."); }
  Serial.println("\nWiFi OK");

  // Sincroniza la hora con servidores NTP
  configTime(GMT_OFFSET, DST_OFFSET, "pool.ntp.org", "time.nist.gov");
}

void loop() {
  struct tm t;
  if (!getLocalTime(&t)) { delay(500); return; }

  // Texto de la hora actual HH:MM
  char hora[6];
  strftime(hora, sizeof(hora), "%H:%M", &t);

  // Busca la proxima dosis del dia para mostrarla
  char prox[6] = "--:--";
  int ahora = t.tm_hour * 60 + t.tm_min;
  int mejor = 24 * 60 + 1;
  for (int i = 0; i < N_DOSIS; i++) {
    int m = HORAS[i] * 60 + MINUTOS[i];
    if (m >= ahora && m < mejor) {
      mejor = m;
      snprintf(prox, sizeof(prox), "%02d:%02d", HORAS[i], MINUTOS[i]);
    }
  }

  // Boton: dosis manual
  if (digitalRead(PIN_BTN) == LOW) {
    delay(30);
    if (digitalRead(PIN_BTN) == LOW) {
      dispensar("manual");
      while (digitalRead(PIN_BTN) == LOW) delay(10);
    }
  }

  // Dosis automatica si coincide hora y minuto (solo una vez por minuto)
  for (int i = 0; i < N_DOSIS; i++) {
    if (t.tm_hour == HORAS[i] && t.tm_min == MINUTOS[i] && ultimaDosisMin != ahora) {
      dispensar("programada");
      ultimaDosisMin = ahora;
    }
  }

  pantalla(hora, prox);
  delay(300);
}
Cómo usar
1
Asegúrate de tener las librerías Adafruit SSD1306 y Adafruit GFX. WiFi.h, time.h y Stepper.h ya vienen con el core del ESP-32.
2
Edita SSID y PASS con tu red, y ajusta GMT_OFFSET a tu huso horario (en segundos: −5 h = -5 * 3600).
3
Define tus horarios en los arreglos HORAS[] y MINUTOS[] (y N_DOSIS si cambias la cantidad).
4
Calibra PASOS_DOSIS: súbelo o bájalo hasta que cada giro suelte una ración del tamaño deseado.
5
Sube el código. La OLED mostrará la hora, la próxima dosis y el total de raciones. Pulsa el botón para una porción manual.
Por qué NTP: el ESP-32 no tiene reloj con batería, así que al reiniciarse perdería la hora. Tomándola de internet con configTime() siempre está exacta. La variable ultimaDosisMin impide que, como el loop() corre varias veces por minuto, la dosis se repita una y otra vez durante ese minuto.

Mejoras: añade un buzzer que avise al dispensar, o sube el conteo de raciones a internet. Si no quieres WiFi, puedes disparar las dosis con un millis() cada X horas, aunque perderías la precisión de la hora real.
