Descripción
Principiante Pantallas · Sensores · I2C

El acelerómetro GY-6500 (MPU-6050) mide la inclinación respecto a la gravedad. La pantalla OLED muestra los ángulos de Pitch y Roll en tiempo real con valores numéricos y barras indicadoras. Cuando el dispositivo está nivelado aparece el mensaje NIVELADO!. Ambos módulos comparten el mismo bus I2C.

GY-6500 — VCC 3.3V —
GY-6500 — GND GND —
I2C compartido: OLED (dirección 0x3C) y GY-6500/MPU-6050 (dirección 0x68) comparten los mismos 2 cables SDA/SCL sin conflicto. Solo necesitas 4 hilos en total para ambos módulos.
Código Arduino
C++ / Arduino IDE (ESP-32) 
#include <Wire.h>
#include <Adafruit_GFX.h>
#include <Adafruit_SSD1306.h>
#include <math.h>

#define SCREEN_W   128
#define SCREEN_H    64
#define OLED_RESET  -1
#define MPU_ADDR   0x68

Adafruit_SSD1306 oled(SCREEN_W, SCREEN_H, &Wire, OLED_RESET);

void mpuInit() {
  Wire.beginTransmission(MPU_ADDR);
  Wire.write(0x6B);  // Registro PWR_MGMT_1
  Wire.write(0x00);  // Despertar el sensor (por defecto esta en sleep)
  Wire.endTransmission(true);
}

void readAccel(float &ax, float &ay, float &az) {
  Wire.beginTransmission(MPU_ADDR);
  Wire.write(0x3B);              // ACCEL_XOUT_H
  Wire.endTransmission(false);
  Wire.requestFrom(MPU_ADDR, 6, true);

  uint8_t d[6];
  for (int i = 0; i < 6; i++) d[i] = Wire.read();

  ax = (int16_t)(d[0] << 8 | d[1]) / 16384.0f;  // Rango +-2g (16384 LSB/g)
  ay = (int16_t)(d[2] << 8 | d[3]) / 16384.0f;
  az = (int16_t)(d[4] << 8 | d[5]) / 16384.0f;
}

void setup() {
  Serial.begin(115200);
  Wire.begin();   // SDA=GPIO21, SCL=GPIO22
  mpuInit();

  if (!oled.begin(SSD1306_SWITCHCAPVCC, 0x3C)) {
    Serial.println("Error: OLED no encontrado");
    while (true);
  }

  oled.setTextColor(SSD1306_WHITE);
  oled.clearDisplay();
  oled.setTextSize(1);
  oled.setCursor(18, 20);
  oled.print("Nivel Digital");
  oled.setCursor(10, 34);
  oled.print("GY-6500 / MPU-6050");
  oled.display();
  delay(1500);
}

void loop() {
  float ax, ay, az;
  readAccel(ax, ay, az);

  // Filtro paso-bajo para suavizar las lecturas
  static float pf = 0.0f, rf = 0.0f;
  float pitch = atan2f( ax, sqrtf(ay*ay + az*az)) * 180.0f / M_PI;
  float roll  = atan2f( ay, sqrtf(ax*ax + az*az)) * 180.0f / M_PI;
  pf = pf * 0.75f + pitch * 0.25f;
  rf = rf * 0.75f + roll  * 0.25f;

  Serial.printf("Pitch: %+.1f  Roll: %+.1f\n", pf, rf);

  oled.clearDisplay();

  // Etiquetas
  oled.setTextSize(1);
  oled.setCursor(4,  0); oled.print("PITCH");
  oled.setCursor(72, 0); oled.print("ROLL");

  // Valores numericos grandes
  char pb[8], rb[8];
  sprintf(pb, "%+.1f", pf);
  sprintf(rb, "%+.1f", rf);
  oled.setTextSize(2);
  oled.setCursor(0,  9); oled.print(pb);
  oled.setCursor(66, 9); oled.print(rb);

  // Separador
  oled.drawLine(0, 27, 127, 27, SSD1306_WHITE);

  // Barra de Pitch (P a la izquierda, barra con marcador central)
  oled.setTextSize(1);
  oled.setCursor(0, 32); oled.print("P");
  oled.drawRect(12, 31, 114, 8, SSD1306_WHITE);
  oled.drawLine(69, 31, 69, 38, SSD1306_WHITE);
  int pd = 69 + (int)constrain(pf / 45.0f * 53.0f, -53.0f, 53.0f);
  oled.fillRect(pd - 2, 32, 5, 6, SSD1306_WHITE);

  // Barra de Roll
  oled.setCursor(0, 44); oled.print("R");
  oled.drawRect(12, 43, 114, 8, SSD1306_WHITE);
  oled.drawLine(69, 43, 69, 50, SSD1306_WHITE);
  int rd = 69 + (int)constrain(rf / 45.0f * 53.0f, -53.0f, 53.0f);
  oled.fillRect(rd - 2, 44, 5, 6, SSD1306_WHITE);

  // Estado de nivel
  oled.setCursor(14, 56);
  if (fabsf(pf) < 2.5f && fabsf(rf) < 2.5f)
    oled.print("** NIVELADO! **");
  else
    oled.print("Inclina p/ nivelar");

  oled.display();
  delay(50);
}
Cómo usar
1
Instala solo Adafruit SSD1306 desde el Gestor de Librerías. No se necesita ninguna librería extra para el GY-6500 — el código lo lee directamente por I2C con Wire.h.
2
Conecta OLED y GY-6500 al mismo bus I2C (GPIO 21 y GPIO 22). Ambos módulos comparten los mismos 2 pines de datos.
3
Sube el código. La pantalla mostrará los ángulos Pitch y Roll actualizados 20 veces por segundo.
4
Coloca el protoboard sobre una superficie plana. El mensaje NIVELADO! aparece cuando ambos ángulos están dentro de ±2.5°.
5
Las barras P y R muestran el ángulo visualmente: el marcador en el centro = nivelado. Rango ±45° representado en la barra.
Tip: Si los ejes Pitch y Roll están invertidos respecto a la orientación física de tu módulo, intercambia ax y ay en las líneas de cálculo de pitch y roll. La orientación del chip MPU-6050 en el PCB varía según el fabricante.
