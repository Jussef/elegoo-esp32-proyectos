← Volver / OLED 02 / Medidor de Distancia
Descripción
Principiante Pantallas · Sensores

El sensor ultrasónico HC-SR04 emite un pulso de sonido y mide el tiempo que tarda en rebotar. La pantalla OLED muestra la distancia en centímetros en tiempo real con un número grande y una barra de proximidad: cuanto más llena, más cerca está el objeto. Rango útil: 2 cm a 4 metros.

OLED — GND GND —
HC-SR04 — VCC 3.3V ⚠ Usar 3.3V (no 5V) para proteger el pin ECHO del ESP32
HC-SR04 — TRIG GPIO 5 Salida digital del ESP32
HC-SR04 — ECHO GPIO 18 Entrada digital — seguro a 3.3V
HC-SR04 — GND GND —
Importante: El HC-SR04 alimentado desde 5V genera una señal de 5V en el pin ECHO, lo que puede dañar el ESP32 (máx. 3.3V). Alimentándolo desde 3.3V el ECHO también sale a 3.3V y el sensor funciona correctamente a esa tensión.
Código Arduino
C++ / Arduino IDE (ESP-32) 
#include <Wire.h>
#include <Adafruit_GFX.h>
#include <Adafruit_SSD1306.h>

#define SCREEN_W   128
#define SCREEN_H    64
#define OLED_RESET  -1
#define TRIG_PIN     5
#define ECHO_PIN    18

Adafruit_SSD1306 oled(SCREEN_W, SCREEN_H, &Wire, OLED_RESET);

long medirCm() {
  digitalWrite(TRIG_PIN, LOW);
  delayMicroseconds(2);
  digitalWrite(TRIG_PIN, HIGH);
  delayMicroseconds(10);
  digitalWrite(TRIG_PIN, LOW);
  long us = pulseIn(ECHO_PIN, HIGH, 25000UL);  // timeout 25 ms = ~4 m
  return (us == 0) ? -1 : us * 17 / 1000;      // us * 0.017 cm/us
}

void setup() {
  Serial.begin(115200);
  pinMode(TRIG_PIN, OUTPUT);
  pinMode(ECHO_PIN, INPUT);

  if (!oled.begin(SSD1306_SWITCHCAPVCC, 0x3C)) {
    Serial.println("Error: OLED no encontrado");
    while (true);
  }

  oled.setTextColor(SSD1306_WHITE);
  oled.clearDisplay();
  oled.setTextSize(1);
  oled.setCursor(12, 28);
  oled.print("Medidor HC-SR04");
  oled.display();
  delay(1500);
}

void loop() {
  long dist = medirCm();

  Serial.print("Distancia: ");
  Serial.print(dist);
  Serial.println(" cm");

  oled.clearDisplay();

  // Titulo
  oled.setTextSize(1);
  oled.setCursor(28, 1);
  oled.print("DISTANCIA");
  oled.drawLine(0, 11, 127, 11, SSD1306_WHITE);

  if (dist < 0 || dist > 400) {
    oled.setCursor(14, 26);
    oled.print("Sin objeto cercano");
    oled.setCursor(30, 40);
    oled.print("> 4 metros");
  } else {
    // Valor grande centrado
    char buf[10];
    sprintf(buf, "%ld cm", dist);
    oled.setTextSize(3);
    int tw = strlen(buf) * 18;
    int cx = (128 - tw) / 2;
    if (cx < 0) cx = 0;
    oled.setCursor(cx, 15);
    oled.print(buf);

    // Etiquetas de barra
    oled.setTextSize(1);
    oled.setCursor(2, 44);
    oled.print("Lejos");
    oled.setCursor(86, 44);
    oled.print("Cerca");

    // Barra de proximidad: mas cerca = barra mas llena (izq -> der)
    oled.drawRect(2, 54, 124, 8, SSD1306_WHITE);
    if (dist < 100) {
      long fill = map(dist, 0, 100, 124, 0);
      oled.fillRect(2, 54, (int)fill, 8, SSD1306_WHITE);
    }
  }

  oled.display();
  delay(150);
}
Cómo usar
1
En el Gestor de Librerías de Arduino IDE instala Adafruit SSD1306 (instalará Adafruit GFX como dependencia automáticamente).
2
Conecta el OLED y el HC-SR04 según la tabla. Importante: alimenta el HC-SR04 desde el pin 3.3V del ESP32, no desde el de 5V.
3
Sube el código y apunta el sensor a distintos objetos. La distancia se refresca cada 150 ms. También puedes verla en el Monitor Serie a 115200 baud.
4
La barra inferior se llena a medida que un objeto se acerca a 1 metro. Para 0–100 cm la barra refleja la proximidad en tiempo real.
Tip: Si ves lecturas erráticas, aumenta el delay(150) a delay(300). El sensor necesita tiempo para que el eco anterior se disipe antes de la siguiente medición, especialmente en espacios cerrados o con superficies muy reflectantes.
