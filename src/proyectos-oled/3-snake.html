Descripción
Avanzado Pantallas · Juegos

El clásico juego Snake corriendo en la pantalla OLED de 128×64 píxeles. El joystick controla la dirección de la serpiente. Cada comida que tocas suma un punto y el juego se acelera gradualmente. El botón del joystick inicia y reinicia la partida. Grid de 32×16 celdas, hasta 200 segmentos de longitud.

Mecánica del juego
Grid
32 × 16 celdas (4 px/celda)
Velocidad inicial
180 ms por movimiento
Aceleración
−8 ms por comida (mín. 70 ms)
Game Over
Pared o colisión propia
Componentes
ESP-32 Board1 PC
0.96" OLED Display (I2C)1 PC
Joystick Module1 PC
Protoboard + cables Dupont—
Conexiones
ComponentePin ESP-32Notas
OLED — SDA GPIO 21 I2C datos
OLED — SCL GPIO 22 I2C reloj
OLED — VCC 3.3V —
OLED — GND GND —
Joystick — VRx GPIO 34 Analógico X (pin solo-entrada, ideal para ADC)
Joystick — VRy GPIO 35 Analógico Y (pin solo-entrada, ideal para ADC)
Joystick — SW GPIO 15 Botón del joystick (INPUT_PULLUP)
Joystick — +5V 3.3V ⚠ Usar 3.3V para que los analógicos no superen el ADC del ESP32
Joystick — GND GND —
Importante: Conecta el VCC del joystick a 3.3V y no a 5V. Las salidas analógicas VRx/VRy llegan hasta la tensión de alimentación; si alimentas con 5V los pines ADC del ESP32 recibirían 5V, dañando el microcontrolador.
Código Arduino
C++ / Arduino IDE (ESP-32) 
#include <Wire.h>
#include <Adafruit_GFX.h>
#include <Adafruit_SSD1306.h>

#define SCREEN_W   128
#define SCREEN_H    64
#define OLED_RESET  -1

#define JOY_X   34
#define JOY_Y   35
#define JOY_SW  15

#define CELL     4
#define COLS    32    // SCREEN_W / CELL
#define ROWS    16    // SCREEN_H / CELL
#define MAX_LEN 200

Adafruit_SSD1306 oled(SCREEN_W, SCREEN_H, &Wire, OLED_RESET);

struct Pt { int8_t x, y; };

Pt     snake[MAX_LEN];
int    snakeLen;
Pt     food;
int8_t dx, dy;
int    score;
bool   gameOver;
unsigned long lastMove;
int    speed;

void spawnFood() {
  bool ok;
  do {
    ok = true;
    food = { (int8_t)random(COLS), (int8_t)random(ROWS) };
    for (int i = 0; i < snakeLen; i++)
      if (snake[i].x == food.x && snake[i].y == food.y) { ok = false; break; }
  } while (!ok);
}

void initGame() {
  snakeLen = 3;
  snake[0] = { 16, 8 };
  snake[1] = { 15, 8 };
  snake[2] = { 14, 8 };
  dx = 1; dy = 0;
  score = 0;
  gameOver = false;
  speed = 180;
  lastMove = 0;
  spawnFood();
}

void drawScene() {
  oled.clearDisplay();

  for (int i = 0; i < snakeLen; i++) {
    int px = snake[i].x * CELL;
    int py = snake[i].y * CELL;
    if (i == 0)
      oled.fillRect(px, py, CELL, CELL, SSD1306_WHITE);           // cabeza: celda completa
    else
      oled.fillRect(px, py, CELL - 1, CELL - 1, SSD1306_WHITE);  // cuerpo: con espacio
  }

  // Comida: cuadro 2x2 centrado en la celda
  oled.fillRect(food.x * CELL + 1, food.y * CELL + 1, 2, 2, SSD1306_WHITE);

  oled.display();
}

bool btnPressed() {
  if (digitalRead(JOY_SW) == LOW) {
    delay(50);
    while (digitalRead(JOY_SW) == LOW);
    delay(50);
    return true;
  }
  return false;
}

void showStart() {
  oled.clearDisplay();
  oled.setTextColor(SSD1306_WHITE);
  oled.setTextSize(2);
  oled.setCursor(22, 10);
  oled.print("SNAKE");
  oled.setTextSize(1);
  oled.setCursor(10, 40);
  oled.print("Presiona el boton");
  oled.display();
}

void showGameOver() {
  oled.clearDisplay();
  oled.setTextColor(SSD1306_WHITE);
  oled.setTextSize(2);
  oled.setCursor(10, 6);
  oled.print("GAME OVER");
  oled.drawLine(0, 27, 127, 27, SSD1306_WHITE);
  oled.setTextSize(1);
  oled.setCursor(18, 34);
  oled.print("Puntuacion: ");
  oled.print(score);
  oled.setCursor(8, 50);
  oled.print("Boton para reiniciar");
  oled.display();
}

void setup() {
  Serial.begin(115200);
  pinMode(JOY_SW, INPUT_PULLUP);
  randomSeed(analogRead(36) + millis());

  if (!oled.begin(SSD1306_SWITCHCAPVCC, 0x3C)) {
    Serial.println("Error: OLED no encontrado");
    while (true);
  }

  showStart();
  while (!btnPressed()) delay(20);
  initGame();
}

void loop() {
  if (gameOver) {
    showGameOver();
    while (!btnPressed()) delay(50);
    initGame();
    return;
  }

  // Leer joystick
  int jx = analogRead(JOY_X);
  int jy = analogRead(JOY_Y);

  if      (jx < 1000 && dy != 0) { dx = -1; dy =  0; }
  else if (jx > 3000 && dy != 0) { dx =  1; dy =  0; }
  else if (jy < 1000 && dx != 0) { dx =  0; dy = -1; }
  else if (jy > 3000 && dx != 0) { dx =  0; dy =  1; }

  unsigned long now = millis();
  if (now - lastMove < (unsigned long)speed) { drawScene(); return; }
  lastMove = now;

  // Nueva cabeza
  Pt head = { (int8_t)(snake[0].x + dx), (int8_t)(snake[0].y + dy) };

  // Colision con pared
  if (head.x < 0 || head.x >= COLS || head.y < 0 || head.y >= ROWS) {
    gameOver = true; return;
  }

  // Colision consigo mismo
  for (int i = 0; i < snakeLen - 1; i++) {
    if (snake[i].x == head.x && snake[i].y == head.y) {
      gameOver = true; return;
    }
  }

  bool ate = (head.x == food.x && head.y == food.y);

  // Mover o crecer
  if (ate && snakeLen < MAX_LEN) snakeLen++;
  for (int i = snakeLen - 1; i > 0; i--) snake[i] = snake[i - 1];
  snake[0] = head;

  if (ate) {
    score++;
    if (speed > 70) speed -= 8;
    spawnFood();
    Serial.print("Score: ");
    Serial.println(score);
  }

  drawScene();
}
Cómo usar
1
Instala Adafruit SSD1306 desde el Gestor de Librerías de Arduino IDE (instalará Adafruit GFX automáticamente como dependencia).
2
Conecta el OLED y el joystick según la tabla. El VCC del joystick va a 3.3V, no a 5V.
3
Sube el código. La pantalla mostrará "SNAKE — Presiona el botón". Presiona el stick para comenzar.
4
Mueve el joystick en las 4 direcciones para controlar la serpiente. No puedes girar 180° de golpe (sin retroceso directo).
5
Cada comida suma un punto y reduce el intervalo en 8 ms. Cuando chocas con la pared o contigo mismo aparece la pantalla Game Over. Presiona el botón para reiniciar.
Ajuste de dificultad: Cambia speed = 180 por un valor más bajo (ej. 120) para empezar más rápido, o más alto (ej. 250) para una dificultad menor. El mínimo está fijado en 70 ms (límite físico del display para refrescar a tiempo).
