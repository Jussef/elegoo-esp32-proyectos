Descripción
Principiante Pantallas · WiFi · Sensores

El ESP-32 se conecta a internet y sincroniza la hora exacta con servidores NTP. La pantalla OLED muestra el reloj en tiempo real: hora, día de la semana y fecha completa. El sensor DHT11 añade temperatura y humedad actualizadas cada segundo en la parte inferior.

Código Arduino
C++ / Arduino IDE (ESP-32) 
#include <Wire.h>
#include <Adafruit_GFX.h>
#include <Adafruit_SSD1306.h>
#include <DHT.h>
#include <WiFi.h>
#include <time.h>

#define SCREEN_W   128
#define SCREEN_H    64
#define OLED_RESET  -1
#define DHT_PIN      4

const char* SSID     = "Tu_Red_WiFi";
const char* PASSWORD = "Tu_Contrasena";
const long  UTC_OFFSET = -18000;   // UTC-5 (Mexico/Bogota/Lima). UTC-3 = -10800

const char* DIAS[] = { "Dom", "Lun", "Mar", "Mie", "Jue", "Vie", "Sab" };

Adafruit_SSD1306 oled(SCREEN_W, SCREEN_H, &Wire, OLED_RESET);
DHT dht(DHT_PIN, DHT11);

void setup() {
  Serial.begin(115200);
  dht.begin();

  if (!oled.begin(SSD1306_SWITCHCAPVCC, 0x3C)) {
    Serial.println("Error: OLED no encontrado");
    while (true);
  }

  oled.clearDisplay();
  oled.setTextColor(SSD1306_WHITE);
  oled.setTextSize(1);
  oled.setCursor(12, 24);
  oled.print("Conectando WiFi...");
  oled.display();

  WiFi.begin(SSID, PASSWORD);
  while (WiFi.status() != WL_CONNECTED) delay(500);

  configTime(UTC_OFFSET, 0, "pool.ntp.org");

  struct tm t;
  while (!getLocalTime(&t)) delay(500);

  oled.clearDisplay();
  oled.setCursor(10, 24);
  oled.print("Listo! Sincronizado.");
  oled.display();
  delay(1200);
}

void loop() {
  struct tm t;
  if (!getLocalTime(&t)) return;

  float temp = dht.readTemperature();
  float hum  = dht.readHumidity();

  oled.clearDisplay();

  // Hora en texto grande
  char hora[9];
  strftime(hora, sizeof(hora), "%H:%M:%S", &t);
  oled.setTextSize(2);
  oled.setCursor(4, 2);
  oled.print(hora);

  oled.drawLine(0, 22, 127, 22, SSD1306_WHITE);

  // Dia y fecha
  char fecha[11];
  strftime(fecha, sizeof(fecha), "%d/%m/%Y", &t);
  oled.setTextSize(1);
  oled.setCursor(0, 26);
  oled.print(DIAS[t.tm_wday]);
  oled.print("  ");
  oled.print(fecha);

  oled.drawLine(0, 38, 127, 38, SSD1306_WHITE);

  // Temperatura
  oled.setCursor(0, 42);
  if (!isnan(temp)) {
    oled.print("Temp: ");
    oled.print(temp, 1);
    oled.print(" C");
  } else {
    oled.print("Temp: -- C");
  }

  // Humedad
  oled.setCursor(0, 54);
  if (!isnan(hum)) {
    oled.print("Hum:  ");
    oled.print((int)hum);
    oled.print(" %");
  } else {
    oled.print("Hum:  -- %");
  }

  oled.display();
  delay(1000);
}
Cómo usar
1
En Arduino IDE abre el Gestor de Librerías y busca e instala Adafruit SSD1306 (instala también Adafruit GFX cuando te lo pida) y DHT sensor library de Adafruit.
2
Conecta la pantalla OLED y el DHT11 según la tabla. El OLED usa solo 4 cables (I2C): SDA, SCL, VCC y GND.
3
Edita SSID y PASSWORD con los datos de tu red WiFi doméstica.
4
Ajusta UTC_OFFSET a tu zona horaria: -18000 para UTC-5 (México, Colombia, Perú), -10800 para UTC-3 (Argentina).
5
Sube el código. La pantalla mostrará "Conectando WiFi..." y luego comenzará a mostrar el reloj con datos del sensor.
Tip: Si la pantalla no aparece, verifica la dirección I2C de tu OLED. La mayoría usa 0x3C pero algunas usan 0x3D. Puedes usar el ejemplo I2C Scanner de Arduino para confirmarlo.
