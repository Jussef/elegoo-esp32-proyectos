Calculadora funcional de 4 operaciones usando el teclado matricial 4×4 y la pantalla OLED. La línea superior muestra la expresión en curso y la zona inferior muestra el número actual en tamaño grande. Soporta encadenamiento de operaciones, números decimales y detecta la división por cero.

Código Arduino
C++ / Arduino IDE (ESP-32) 
#include <Wire.h>
#include <Adafruit_GFX.h>
#include <Adafruit_SSD1306.h>
#include <Keypad.h>
#include <math.h>

#define SCREEN_W   128
#define SCREEN_H    64
#define OLED_RESET  -1

const byte KROWS = 4, KCOLS = 4;
char keys[KROWS][KCOLS] = {
  {'1','2','3','A'},
  {'4','5','6','B'},
  {'7','8','9','C'},
  {'*','0','#','D'}
};
byte rowPins[KROWS] = {13, 14, 27, 26};
byte colPins[KCOLS] = {25, 33, 32,  4};

Keypad           keypad = Keypad(makeKeymap(keys), rowPins, colPins, KROWS, KCOLS);
Adafruit_SSD1306 oled(SCREEN_W, SCREEN_H, &Wire, OLED_RESET);

String currentInput = "0";
String displayVal   = "0";
String exprLine     = "";
float  op1          = 0;
char   pendingOp    = 0;
bool   freshInput   = true;

// Convierte float a String eliminando decimales innecesarios
String fmtNum(float v) {
  if (!isfinite(v)) return "Error";
  if (v == (float)(long)v && fabsf(v) < 1e9f) return String((long)v);
  String s = String(v, 4);
  while (s.indexOf('.') >= 0 && s.endsWith("0")) s.remove(s.length() - 1);
  if (s.endsWith(".")) s.remove(s.length() - 1);
  return s;
}

String opSym(char op) { return (op == '*') ? "x" : String(op); }

void showOLED() {
  oled.clearDisplay();

  // Linea de expresion (arriba, texto pequeno)
  oled.setTextSize(1);
  oled.setCursor(2, 2);
  String expr = (exprLine.length() > 21) ? exprLine.substring(exprLine.length() - 21) : exprLine;
  oled.print(expr);
  oled.drawLine(0, 13, 127, 13, SSD1306_WHITE);

  // Valor actual (abajo, texto grande y centrado)
  int tsize = (displayVal.length() <= 5) ? 3 : (displayVal.length() <= 9) ? 2 : 1;
  int tw    = displayVal.length() * 6 * tsize;
  int textH = 8 * tsize;
  int cx    = max(2, (128 - tw) / 2);
  int cy    = 14 + (50 - textH) / 2;
  oled.setTextSize(tsize);
  oled.setCursor(cx, cy);
  oled.print(displayVal);

  oled.display();
}

void handleKey(char key) {
  if (key >= '0' && key <= '9') {
    if (freshInput || currentInput == "0") {
      currentInput = (key == '0') ? "0" : String(key);
      freshInput = false;
    } else if (currentInput.length() < 11) {
      currentInput += key;
    }
    displayVal = currentInput;
    exprLine = pendingOp ? (fmtNum(op1) + " " + opSym(pendingOp) + " " + currentInput)
                         : currentInput;

  } else if (key == 'A' || key == 'B' || key == 'C' || key == 'D') {
    char ops[] = {'+', '-', '*', '/'};
    op1          = currentInput.toFloat();
    pendingOp    = ops[key - 'A'];
    freshInput   = true;
    currentInput = "0";
    displayVal   = "0";
    exprLine     = fmtNum(op1) + " " + opSym(pendingOp);

  } else if (key == '#') {  // Igual =
    if (pendingOp && !freshInput) {
      float op2 = currentInput.toFloat();
      float res;
      switch (pendingOp) {
        case '+': res = op1 + op2; break;
        case '-': res = op1 - op2; break;
        case '*': res = op1 * op2; break;
        case '/': res = (op2 != 0.0f) ? op1 / op2 : NAN; break;
        default:  res = op2;
      }
      exprLine   = fmtNum(op1) + " " + opSym(pendingOp) + " " + fmtNum(op2) + " =";
      displayVal = isfinite(res) ? fmtNum(res) : "Error /0";
      if (isfinite(res)) { op1 = res; currentInput = displayVal; }
      pendingOp  = 0;
      freshInput = true;
    }

  } else if (key == '*') {  // C = Borrar
    currentInput = "0"; displayVal = "0"; exprLine = "";
    op1 = 0; pendingOp = 0; freshInput = true;
  }

  showOLED();
}

void setup() {
  Serial.begin(115200);
  if (!oled.begin(SSD1306_SWITCHCAPVCC, 0x3C)) {
    Serial.println("Error: OLED no encontrado");
    while (true);
  }
  oled.setTextColor(SSD1306_WHITE);
  showOLED();
}

void loop() {
  char key = keypad.getKey();
  if (key) {
    Serial.print("Tecla: "); Serial.println(key);
    handleKey(key);
  }
}
Cómo usar
1
Instala Adafruit SSD1306 y la librería Keypad de Mark Stanley & Alexander Brevig desde el Gestor de Librerías de Arduino IDE.
2
Conecta el OLED por I2C (pines 21/22) y el teclado matricial según la tabla. El conector del keypad tiene 8 pines; conecta en orden de izquierda a derecha.
3
Sube el código. La pantalla muestra 0 listo para ingresar el primer número.
4
Escribe un número, presiona A/B/C/D para elegir operación, escribe el segundo número y presiona # para ver el resultado.
5
Presiona * en cualquier momento para borrar y empezar de nuevo. Puedes encadenar operaciones: resultado → operador → número → igual.
Tip: Si alguna tecla no responde, verifica el orden de los pines en el conector del keypad. Puedes usar el Monitor Serie (115200 baud) para ver qué tecla detecta el código con cada pulsación y así diagnosticar problemas de cableado.
